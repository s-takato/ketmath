\newslide{$x^n$の微分}

\vspace*{18mm}

\slidepage
\begin{itemize}
\item
$(c)'=0$（$c$は定数）
\item
$(x)'=1$
\item
$(x^2)'=2x$
\item
$(x^3)'=3x^2$
\item
一般に　$(x^n)'=n x^{n-1}$\\
　・$n$は負の整数でも分数(実数)でもよい
\end{itemize}
